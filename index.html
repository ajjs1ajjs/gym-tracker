Будемо завантажувати з localStorage let completionState = {}; // Функція для збереження прогресу в localStorage function saveState() { localStorage.setItem('trainingProgress', JSON.stringify(completionState)); } // Функція для завантаження прогресу function loadState() { const savedState = localStorage.getItem('trainingProgress'); if (savedState) { completionState = JSON.parse(savedState); } } // Функція для відображення карток вправ function renderExercises() { appContainer.innerHTML = ''; // Очищуємо контейнер trainingData.forEach(muscleGroup => { const groupTitle = document.createElement('h2'); groupTitle.textContent = muscleGroup.group; appContainer.appendChild(groupTitle); const exercisesWrapper = document.createElement('div'); exercisesWrapper.className = 'exercises-wrapper'; muscleGroup.exercises.forEach(exercise => { const isCompleted = completionState[exercise.id] === true; const card = document.createElement('div'); card.className = `exercise-card ${isCompleted ? 'completed' : ''}`; card.innerHTML = `

${exercise.name}

${exercise.description}

`; exercisesWrapper.appendChild(card); }); appContainer.appendChild(exercisesWrapper); }); renderProgress(); } // Функція для оновлення прогресу function renderProgress() { progressContainer.innerHTML = ''; trainingData.forEach(muscleGroup => { const allExercises = muscleGroup.exercises.length; const completedExercises = muscleGroup.exercises.filter(ex => completionState[ex.id]).length; const progressElement = document.createElement('div'); progressElement.className = 'progress-item'; if (completedExercises === 0) { progressElement.classList.add('not-started'); progressElement.textContent = `❌ ${muscleGroup.group} (не зроблено)`; } else if (completedExercises < allExercises) { progressElement.textContent=`⏳ ${muscleGroup.group} (/)`; } else { progressElement.classList.add('fully-completed'); progressElement.textContent=`✅ ${muscleGroup.group} (завершено)`; } progressContainer.appendChild(progressElement); }); } // Обробник подій для "чекінів" appContainer.addEventListener('change', (event)=> { if (event.target.type === 'checkbox') { const exerciseId = event.target.dataset.id; completionState[exerciseId] = event.target.checked; // Зберігаємо та оновлюємо вигляд saveState(); renderExercises(); } }); // Ініціалізація додатку function init() { loadState(); renderExercises(); } init();
